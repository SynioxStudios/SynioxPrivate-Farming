local function formatNumber(num)
    if num >= 1e15 then return string.format("%.2fQ", num/1e15) end
    if num >= 1e12 then return string.format("%.2fT", num/1e12) end
    if num >= 1e9 then return string.format("%.2fB", num/1e9) end
    if num >= 1e6 then return string.format("%.2fM", num/1e6) end
    if num >= 1e3 then return string.format("%.2fK", num/1e3) end
    return string.format("%.0f", num)
end

local function unequipAllPets()
    if player:FindFirstChild("petsFolder") then
        for _, folder in pairs(player.petsFolder:GetChildren()) do
            for _, pet in pairs(folder:GetChildren()) do
                rEvents.equipPetEvent:FireServer("unequipPet", pet)
            end
        end
    end
end

local function equipUniquePet(name)
    if player:FindFirstChild("petsFolder") and player.petsFolder:FindFirstChild("Unique") then
        for _, pet in pairs(player.petsFolder.Unique:GetChildren()) do
            if pet.Name == name then
                rEvents.equipPetEvent:FireServer("equipPet", pet)
                task.wait(0.01) 
            end
        end
    end
end

local function findMachine(machineName)
    local machine = workspace:FindFirstChild("machinesFolder") and workspace.machinesFolder:FindFirstChild(machineName)
    if not machine then
        for _, f in pairs(workspace:GetChildren()) do
            if f:IsA("Folder") and f.Name:find("machines") then
                machine = f:FindFirstChild(machineName)
                if machine then break end
            end
        end
    end
    return machine
end

local function pressE()
    game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.E, false, game)
    task.wait(0.05)
    game:GetService("VirtualInputManager"):SendKeyEvent(false, Enum.KeyCode.E, false, game)
end

local RebTab = window:AddTab("Rebirth Farm")
local timeLabel = RebTab:AddLabel("Time: 0d 0h 0m 0s")
local rebStatsLabel = RebTab:AddLabel("Rebirths: 0 | Gained: 0")
local paceLabel = RebTab:AddLabel("Pace: 0 / Hour | 0 / Day | 0 / Week")

local lastRebirthTime = tick()
local lastRebirthValue = rebirthsStat.Value

rebirthsStat:GetPropertyChangedSignal("Value"):Connect(function()
    local now = tick()
    local gained = rebirthsStat.Value - lastRebirthValue
    if gained > 0 then
        local avgTime = (now - lastRebirthTime) / gained
        local paceH = 3600 / avgTime
        paceLabel.Text = string.format("Pace: %s/H | %s/D | %s/W", formatNumber(paceH), formatNumber(paceH * 24), formatNumber(paceH * 168))
        lastRebirthTime = now
        lastRebirthValue = rebirthsStat.Value
    end
    rebStatsLabel.Text = string.format("Rebirths: %s | Gained: %s", formatNumber(rebirthsStat.Value), formatNumber(rebirthsStat.Value - initialRebirths))
end)

local FarmingTab = window:AddTab("Strength Farm")

FarmingTab:AddTextBox("Rep Speed", function(value)
    local num = tonumber(value)
    if num and num > 0 then repsPerTick = math.floor(num) end
end, {placeholder = "23"})

local function getPing()
    local stats = game:GetService("Stats")
    local pingStat = stats:FindFirstChild("PerformanceStats") and stats.PerformanceStats:FindFirstChild("Ping")
    return pingStat and pingStat:GetValue() or 0
end

FarmingTab:AddSwitch("ðŸ’¢ Auto Strength Farm", function(state)
    strengthOnly = state
    if state then
        task.spawn(function()
            unequipAllPets()
            task.wait(0.1)
            equipUniquePet("Swift Samurai")
            while strengthOnly do
                local startTick = tick()
                while tick() - startTick < 0.75 and strengthOnly do
                    for i = 1, repsPerTick do muscleEvent:FireServer("rep") end
                    task.wait(0.01)
                end
                if getPing() >= 350 then task.wait(1) end
            end
        end)
    end
end)

task.spawn(function()
    while true do
        local elapsed = tick() - startTime
        local h = math.floor((elapsed % 86400) / 3600)
        local m = math.floor((elapsed % 3600) / 60)
        local s = math.floor(elapsed % 60)
        timeLabel.Text = string.format("Time: %dh %dm %ds", h, m, s)
        sStatsLabel.Text = string.format("Strength: %s | Gained: %s", formatNumber(strengthStat.Value), formatNumber(strengthStat.Value - initialStrength))
        if strengthOnly and elapsed > 0 then
            local pace = (strengthStat.Value - initialStrength) / (elapsed / 3600)
            sPaceLabel.Text = "Strength Pace: " .. formatNumber(pace) .. " / Hour"
        end
        task.wait(1)
    end
end)
