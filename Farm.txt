local function formatNumber(num)
    if num >= 1e15 then return string.format("%.2fQa", num/1e15) end
    if num >= 1e12 then return string.format("%.2fT", num/1e12) end
    if num >= 1e9 then return string.format("%.2fB", num/1e9) end
    if num >= 1e6 then return string.format("%.2fM", num/1e6) end
    if num >= 1e3 then return string.format("%.2fK", num/1e3) end
    return string.format("%.0f", num)
end

local RebirthTab = window:AddTab("Rebirth Farm")
local rb_Running = false
local rb_Start = tick()
local rb_Player = game.Players.LocalPlayer
local rb_Initial = rb_Player.leaderstats.Rebirths.Value

RebirthTab:AddLabel("Rebirth Stats").TextSize = 20
local rb_TimeLabel = RebirthTab:AddLabel("Time: 0d 0h 0m 0s")
local rb_GainLabel = RebirthTab:AddLabel("Gained: 0")

task.spawn(function()
    while true do
        if rb_Running then
            local sec = tick() - rb_Start
            rb_TimeLabel.Text = string.format("Time: %dd %dh %dm %ds", math.floor(sec/86400), math.floor((sec%86400)/3600), math.floor((sec%3600)/60), math.floor(sec%60))
            rb_GainLabel.Text = "Gained: " .. (rb_Player.leaderstats.Rebirths.Value - rb_Initial)
        end
        task.wait(1)
    end
end)

RebirthTab:AddSwitch("Auto Rebirth Farm", function(state)
    rb_Running = state
    if state then
        rb_Start = tick()
        task.spawn(function()
            while rb_Running do
                local myRebs = rb_Player.leaderstats.Rebirths.Value
                local cost = 10000 + (5000 * myRebs)
                if rb_Player:FindFirstChild("ultimatesFolder") and rb_Player.ultimatesFolder:FindFirstChild("Golden Rebirth") then
                    cost = math.floor(cost * (1 - (rb_Player.ultimatesFolder["Golden Rebirth"].Value * 0.1)))
                end
                if rb_Player:FindFirstChild("petsFolder") then
                    for _,f in pairs(rb_Player.petsFolder:GetChildren()) do for _,p in pairs(f:GetChildren()) do game:GetService("ReplicatedStorage").rEvents.equipPetEvent:FireServer("unequipPet", p) end end
                end
                task.wait(0.1)
                for _,p in pairs(rb_Player.petsFolder.Unique:GetChildren()) do if p.Name == "Swift Samurai" then game:GetService("ReplicatedStorage").rEvents.equipPetEvent:FireServer("equipPet", p) end end
                while rb_Running and rb_Player.leaderstats.Strength.Value < cost do
                    for i = 1, 12 do rb_Player.muscleEvent:FireServer("rep") end
                    task.wait(0.01)
                end
                if not rb_Running then break end
                if rb_Player:FindFirstChild("petsFolder") then
                    for _,f in pairs(rb_Player.petsFolder:GetChildren()) do for _,p in pairs(f:GetChildren()) do game:GetService("ReplicatedStorage").rEvents.equipPetEvent:FireServer("unequipPet", p) end end
                end
                task.wait(0.1)
                for _,p in pairs(rb_Player.petsFolder.Unique:GetChildren()) do if p.Name == "Tribal Overlord" then game:GetService("ReplicatedStorage").rEvents.equipPetEvent:FireServer("equipPet", p) end end
                rb_Player.Character.HumanoidRootPart.CFrame = CFrame.new(-8666, 34, 2070)
                task.wait(0.5)
                local retry = 0
                repeat 
                    game:GetService("VirtualInputManager"):SendKeyEvent(true, "E", false, game) task.wait(0.05) game:GetService("VirtualInputManager"):SendKeyEvent(false, "E", false, game)
                    task.wait(0.3) retry = retry + 1 
                until rb_Player.Character.Humanoid.Sit or retry > 10 or not rb_Running
                game:GetService("ReplicatedStorage").rEvents.rebirthRemote:InvokeServer("rebirthRequest")
                task.wait(0.5)
            end
        end)
    end
end)

local StrengthTab = window:AddTab("Strength Farm")
local str_Running = false
local str_Start = tick()
local str_Player = game.Players.LocalPlayer
local str_Initial = str_Player.leaderstats.Strength.Value
local str_Reps = 23

StrengthTab:AddLabel("Strength Stats").TextSize = 20
local str_TimeLabel = StrengthTab:AddLabel("Time: 0d 0h 0m 0s")
local str_PaceLabel = StrengthTab:AddLabel("Pace: 0 / Hour")
local str_GainLabel = StrengthTab:AddLabel("Gained: 0")

task.spawn(function()
    while true do
        if str_Running then
            local sec = tick() - str_Start
            str_TimeLabel.Text = string.format("Time: %dd %dh %dm %ds", math.floor(sec/86400), math.floor((sec%86400)/3600), math.floor((sec%3600)/60), math.floor(sec%60))
            local g = str_Player.leaderstats.Strength.Value - str_Initial
            str_GainLabel.Text = "Gained: " .. formatNumber(g)
            if g > 0 and sec > 0 then str_PaceLabel.Text = "Pace: " .. formatNumber((g/sec)*3600) .. "/H" end
        end
        task.wait(1)
    end
end)
